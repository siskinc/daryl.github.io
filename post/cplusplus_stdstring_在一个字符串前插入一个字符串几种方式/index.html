C++ std::string 在一个字符串前插入一个字符串几种方式

Thu, Jul 25, 2019

[TOC]

1、直接使用字符串相加

std::string a = &quot;hello&quot;;

std::string b = &quot;hello&quot;;

for(int i = 0; i &lt; 100; ++i)

{

	a = b + a;

}

2、使用insert函数

std::string a = &quot;hello&quot;;

for(int i = 0; i &lt; 100; ++i)

{

	a.insert(0, &quot;hello&quot;);

}

比较：通过Quick C++ Benchmarks 可得到结果

static void StringCreation(benchmark::State&amp; state) {

  // Code inside this loop is measured repeatedly

  std::string x = &quot;hello&quot;;

  for (auto _ : state) {

    x.insert(0,&quot;hello&quot;);

    // Make sure the variable is not optimized away by compiler

    benchmark::DoNotOptimize(x);

  }

}

// Register the function as a benchmark

BENCHMARK(StringCreation);



static void StringCopy(benchmark::State&amp; state) {

  // Code before the loop is not measured

  std::string x = &quot;hello&quot;;

  for (auto _ : state) {

    x = &quot;hello&quot; + x;

  }

}

BENCHMARK(StringCopy);
