}

func GetSingle() *Single {

    if single == nil {

        single = &amp;Single{}

    }

    return single

}

    fmt.Printf(&quot;%p\n&quot;, GetSingle())

package main



import (

    &quot;fmt&quot;

    &quot;sync&quot;

    &quot;time&quot;

)



type Single2 struct {



}



var single2 *Single2

var lock *sync.Mutex = &amp;sync.Mutex{}



func GetSingle2() *Single2 {

    lock.Lock()

    defer lock.Unlock()

    if single2 == nil {

        single2 = &amp;Single2{}

    }

    return single2

}



func main() {

    go fmt.Printf(&quot;%p\n&quot;, GetSingle2())

    go fmt.Printf(&quot;%p\n&quot;, GetSingle2())

    go fmt.Printf(&quot;%p\n&quot;, GetSingle2())

    time.Sleep(time.Second)

    fmt.Println(&quot;done&quot;)

}

package main

    &quot;fmt&quot;
