golang自动调用结构体中内嵌字段的Init()函数

Wed, Jan 1, 2020 
package main



import (

    &quot;fmt&quot;

    &quot;reflect&quot;

)



type initial interface {

    Init()

}



func triggerInitials(c interface{}) {

    rv := reflect.ValueOf(c)

    if rv.Kind() == reflect.Ptr {

        rv = rv.Elem()

    }

    for i := 0; i &lt; rv.NumField(); i++ {

        value := rv.Field(i)

        if conf, ok := value.Interface().(initial); ok {

            conf.Init()

        }

    }

}



type A struct {



}



func (a *A) Init() {

    fmt.Println(&quot;Init A&quot;)

}



type B struct {

    AA *A

}



func main() {

    b := B{}

    triggerInitials(b)

}
