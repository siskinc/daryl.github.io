Python远程操作SSH

Thu, Nov 5, 2020

三方组件

paramiko

创建ssh 连接

def connect_ssh(host: str, port: int, username: str, password: str):

    ssh = paramiko.SSHClient()

    key = paramiko.AutoAddPolicy()

    ssh.set_missing_host_key_policy(key)

    ssh.connect(host, port, username, password, timeout=5)

    return ssh

执行命令

def execute(ssh: paramiko.SSHClient):

    stdin, stdout, stderr = ssh.exec_command(&#039;command&#039;)
